import numpy as np
import sounddevice as sd
import random

SAMPLE_RATE = 44100

def envelope_sine(t, duration, power=1.0):
    env = np.sin(np.pi * t / duration)
    return env**power

def note(freq, duration, vibrato_rate=0, vibrato_depth=0):
    t = np.linspace(0, duration, int(SAMPLE_RATE * duration), endpoint=False)
    if vibrato_rate > 0 and vibrato_depth > 0:
        vib = vibrato_depth * np.sin(2 * np.pi * vibrato_rate * t)
    else:
        vib = 0
    waveform = np.sin(2 * np.pi * (freq + vib) * t)
    env = envelope_sine(t, duration, power=1.2)
    return waveform * env

def chirp_note(f0, f1, duration):
    t = np.linspace(0, duration, int(SAMPLE_RATE * duration), endpoint=False)
    curve = np.linspace(0, 1, len(t))
    curve = curve * curve * (3 - 2 * curve)
    freqs = f0 + (f1 - f0) * curve
    waveform = np.sin(2 * np.pi * freqs * t)
    env = envelope_sine(t, duration, power=1.2)
    return waveform * env


def silence(duration):
    return np.zeros(int(SAMPLE_RATE * duration))

def bar(*segments):
    return np.concatenate(segments)

def create_tutor_song():
    beat = 0.20

    # Phrase A: a rising call — simple, memorable
    phrase_a = bar(
        chirp_note(600, 900, 1.5 * beat),
        silence(0.3 * beat),
        chirp_note(750, 950, beat),
        silence(0.4 * beat),
        note(1000, 1.5 * beat),
        silence(0.5 * beat),
    )

    # Phrase B: a gentle falling answer
    phrase_b = bar(
        note(900, beat, vibrato_rate=6, vibrato_depth=8),
        silence(0.2 * beat),
        chirp_note(850, 650, 1.5 * beat),
        silence(0.3 * beat),
        note(600, 2 * beat, vibrato_rate=5, vibrato_depth=6),
        silence(0.6 * beat),
    )

    song = np.concatenate([
        phrase_a,
        phrase_b,
        #phrase_a,   # repeat the call — birds do this
        #phrase_b,
    ])

    song /= np.max(np.abs(song))
    return song
