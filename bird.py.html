import random
import numpy as np
from tutor import note, chirp_note, silence, bar
from audio import save_wav

SAMPLE_RATE = 44100

class Bird:
    def __init__(self, tutor_wave):
        self.tutor_wave = tutor_wave
        self.events = self._random_events()
        self.song = self._render()

    def _random_events(self):
        target_duration = len(self.tutor_wave) / SAMPLE_RATE
        total = 0
        events = []
        while total < target_duration:
            e = self._random_event()
            events.append(e)
            total += e[3] + e[4] if e[0] == 'chirp' else e[2] + e[5]
        return events

    def _render(self, events=None):
        events = events or self.events
        segments = []
        for e in events:
            if e[0] == 'chirp':
                _, f0, f1, dur, gap = e
                segments.append(chirp_note(f0, f1, dur))
            else:
                _, freq, dur, vr, vd, gap = e
                segments.append(note(freq, dur, vr, vd))
            segments.append(silence(gap))
        song = bar(*segments)

        # ── Fix: force song to exactly match tutor length ──────────────────
        target_len = len(self.tutor_wave)
        if len(song) >= target_len:
            song = song[:target_len]  # trim excess
        else:
            repeats = -(-target_len // len(song))  # ceiling division
            song = np.tile(song, repeats)[:target_len]  # loop to fill
        # ──────────────────────────────────────────────────────────────────

        song /= np.max(np.abs(song)) if np.max(np.abs(song)) > 0 else 1
        return song

    def _mutate(self, strength=0.3):
        mutated = list(self.events)

        if random.random() < 0.2 * strength:  # was 0.2
            i = random.randint(0, len(mutated) - 1)
            mutated[i] = self._random_event()

        if random.random() < 0.1 * strength and len(mutated) > 2:  # was 0.1
            mutated.pop(random.randint(0, len(mutated) - 1))
        elif random.random() < 0.1 * strength:
            mutated.insert(random.randint(0, len(mutated)), self._random_event())


        # always nudge parameters
        result = []
        for e in mutated:
            if e[0] == 'chirp':
                _, f0, f1, dur, gap = e
                result.append(('chirp',
                               np.clip(f0 + random.gauss(0, 30 * strength), 200, 1200),
                               np.clip(f1 + random.gauss(0, 30 * strength), 200, 1200),
                               max(0.02, dur + random.gauss(0, 0.01 * strength)),
                               gap))
            else:
                _, freq, dur, vr, vd, gap = e
                result.append(('note',
                               np.clip(freq + random.gauss(0, 30 * strength), 200, 1200),
                               max(0.02, dur + random.gauss(0, 0.01 * strength)),
                               vr, vd, gap))
        return result

    def _random_event(self):
        beat = 0.20
        gap = random.uniform(0.15, 0.6) * beat
        if random.random() < 0.5:
            f0 = random.uniform(550, 800)
            f1 = random.uniform(800, 1100)
            dur = random.uniform(0.8, 1.8) * beat
            return ('chirp', f0, f1, dur, gap)
        else:
            freq = random.uniform(600, 1050)
            dur = random.uniform(0.8, 2.0) * beat
            vr = random.uniform(5, 9) if random.random() < 0.5 else 0
            vd = random.uniform(5, 12) if vr > 0 else 0
            return ('note', freq, dur, vr, vd, gap)  # 6 values: type, freq, dur, vr, vd, gap

    def sing(self):
        return self.song

    def export_audio(self, i):
        save_wav(self.song, f"outputs/bird_{i}.wav")

    def _mutate_structure(self):
        mutated = list(self.events)

        action = random.choice(['swap', 'reverse', 'shuffle', 'duplicate', 'drop'])

        if action == 'swap' and len(mutated) > 1:
            i, j = random.sample(range(len(mutated)), 2)
            mutated[i], mutated[j] = mutated[j], mutated[i]

        elif action == 'reverse':
            i, j = sorted(random.sample(range(len(mutated)), 2))
            mutated[i:j] = mutated[i:j][::-1]

        elif action == 'shuffle':
            random.shuffle(mutated)

        elif action == 'duplicate' and len(mutated) > 1:
            i = random.randint(0, len(mutated) - 1)
            mutated.insert(i, mutated[i])

        elif action == 'drop' and len(mutated) > 2:
            mutated.pop(random.randint(0, len(mutated) - 1))

        return mutated
