def plot_frequency_progression(tutor_wave, steps, sr=44100):
    step_list = list(steps)
    n_plots = len(step_list) + 1

    fig, axes = plt.subplots(n_plots, 1, figsize=(12, 2 * n_plots), sharex=True)

    def plot_f0(ax, wave, title):
        f0 = librosa.yin(wave.astype(float), fmin=200, fmax=1200, sr=sr)
        ax.plot(f0)
        ax.set_title(title)
        ax.set_ylabel("Hz")
        ax.set_ylim(200, 1200)

    plot_f0(axes[0], tutor_wave, "Tutor")

    for idx, i in enumerate(step_list):
        filename = f"outputs/bird_{i}.wav"
        if os.path.exists(filename):
            _, data = read(filename)
            wave = data.astype(float) / 32767
            plot_f0(axes[idx + 1], wave, f"Step {i}")

    plt.xlabel("Frame")
    plt.tight_layout()
    plt.savefig("outputs/frequency_progression.png", dpi=150)
    plt.show()

plot_frequency_progression(tutor_wave, range(0, 500, 100))
