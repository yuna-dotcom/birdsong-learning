import numpy as np
import librosa
import random
from audio import save_wav

SAMPLE_RATE = 44100

def mfcc_error(wave_a, wave_b):
    mfcc_a = librosa.feature.mfcc(y=wave_a.astype(float), sr=SAMPLE_RATE)
    mfcc_b = librosa.feature.mfcc(y=wave_b.astype(float), sr=SAMPLE_RATE)
    # trim to same length
    n = min(mfcc_a.shape[1], mfcc_b.shape[1])
    return np.mean((mfcc_a[:, :n] - mfcc_b[:, :n]) ** 2)


class Learning:
    def __init__(self, tutor_wave, bird):
        self.tutor_wave = tutor_wave
        self.bird = bird
        self.error_history = []

    def run(self, steps=500, export_every=100):
        self.current_error = mfcc_error(self.bird.song, self.tutor_wave)
        for i in range(steps):
            strength = max(0.05, 1.0 * (1 - i / steps))
            self.step(strength)
            if i % export_every == 0:
                self.bird.export_audio(i)
                print(f"Step {i}: error {self.error_history[-1]:.4f}")

    def step(self, strength=0.3, n_candidates=5):
        best_error = self.current_error
        best_events = None
        best_wave = None

        for _ in range(n_candidates):
            # early on: mostly structural, later: mostly parameter tuning
            if random.random() < strength:
                candidate_events = self.bird._mutate_structure()
            else:
                candidate_events = self.bird._mutate(strength)

            candidate_wave = self.bird._render(candidate_events)
            candidate_error = mfcc_error(candidate_wave, self.tutor_wave)
            if candidate_error < best_error:
                best_error = candidate_error
                best_events = candidate_events
                best_wave = candidate_wave

        if best_events is not None:
            self.bird.events = best_events
            self.bird.song = best_wave
            self.current_error = best_error

        self.error_history.append(self.current_error)
